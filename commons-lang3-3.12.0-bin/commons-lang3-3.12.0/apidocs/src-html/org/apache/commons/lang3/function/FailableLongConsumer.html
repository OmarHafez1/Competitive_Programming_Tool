018package org.apache.commons.lang3.function;
019
020import java.util.Objects;
021import java.util.function.LongConsumer;
022
023/**
024 * A functional interface like {@link LongConsumer} that declares a {@code Throwable}.
025 *
026 * @param <E> Thrown exception.
027 * @since 3.11
028 */
029@FunctionalInterface
030public interface FailableLongConsumer<E extends Throwable> {
031
032    /** NOP singleton */
033    @SuppressWarnings("rawtypes")
034    FailableLongConsumer NOP = t -> {/* NOP */};
035
036    /**
037     * Returns The NOP singleton.
038     *
039     * @param <E> Thrown exception.
040     * @return The NOP singleton.
041     */
042    static <E extends Throwable> FailableLongConsumer<E> nop() {
043        return NOP;
044    }
045
046    /**
047     * Accepts the consumer.
048     *
049     * @param object the parameter for the consumable to accept
050     * @throws E Thrown when the consumer fails.
051     */
052    void accept(long object) throws E;
053
054    /**
055     * Returns a composed {@code FailableLongConsumer} like {@link LongConsumer#andThen(LongConsumer)}.
056     *
057     * @param after the operation to perform after this one.
058     * @return a composed {@code FailableLongConsumer} like {@link LongConsumer#andThen(LongConsumer)}.
059     * @throws NullPointerException if {@code after} is null
060     */
061    default FailableLongConsumer<E> andThen(final FailableLongConsumer<E> after) {
062        Objects.requireNonNull(after);
063        return (final long t) -> {
064            accept(t);
065            after.accept(t);
066        };
067    }
068}
