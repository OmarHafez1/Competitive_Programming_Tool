018package org.apache.commons.lang3.function;
019
020import java.util.function.ToIntFunction;
021
022/**
023 * A functional interface like {@link ToIntFunction} that declares a {@code Throwable}.
024 *
025 * @param <T> the type of the argument to the function
026 * @param <E> Thrown exception.
027 * @since 3.11
028 */
029@FunctionalInterface
030public interface FailableToIntFunction<T, E extends Throwable> {
031
032    /** NOP singleton */
033    @SuppressWarnings("rawtypes")
034    FailableToIntFunction NOP = t -> 0;
035
036    /**
037     * Returns The NOP singleton.
038     *
039     * @param <T> the type of the argument to the function
040     * @param <E> Thrown exception.
041     * @return The NOP singleton.
042     */
043    static <T, E extends Throwable> FailableToIntFunction<T, E> nop() {
044        return NOP;
045    }
046
047    /**
048     * Applies this function to the given arguments.
049     *
050     * @param t the first function argument
051     * @return the function result
052     * @throws E Thrown when the function fails.
053     */
054    int applyAsInt(T t) throws E;
055}
